#include<Gl\glut.h>
#include<iostream>
using namespace std;


float x1=100, x2=300,x3,x4,y1=100,y2=300,y3,y4;
float tx = 2,ty =2;
void init(void)
{
    glClearColor(1.0, 1.0, 1.0, 0.0);

    glPointSize(5.0);
    glMatrixMode(GL_PROJECTION);
    gluOrtho2D(0,800,0,800);
}


void rec (float x1, float y1, float x2, float y2)
{
    glBegin(GL_LINE_LOOP);
    glVertex2f(x1, y1);
    glVertex2f(x2, y1);
    glVertex2f(x2, y2);
    glVertex2f(x1, y2);
    glEnd();
}



void display(void)
{
    glClear(GL_COLOR_BUFFER_BIT);
    glColor3f(1.0, 0.0, 0.0);
    //dda(x_s, y_s, x_e, y_e);
    rec(x1,y1,x2,y2);
    glColor3f(0.0, 0.0, 1.0);
    rec(x1*tx, y1*ty, x2*tx, y2*ty);
    glFlush();
}
int main(int argc, char** argv)
{
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(800,800);
    glutInitWindowPosition(300,00);
    glutCreateWindow("Point");
    init();
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
